1. head -n 1 iris.csv
2. tail -n +2 iris.csv
3. cat iris.csv | sed -e 's/Setosa/1/g' -e 's/Versicolor/2/g' -e 's/Virginica/3/g'
4. tail -n +2 iris.csv | cut -d ',' -f 5 | tr -d '"' | sort | uniq
5. awk -F, 'NR>1 {print $1 + $2 + $3 + $4}' iris.csv 
6. awk -F, 'NR>1 {sum += $2} END {print sum / (NR-1)}' iris.csv 
7. awk -F, 'NR>1 {if ($4 > max) {max = $4; line = $0}} END {print line}' iris.csv
8. awk -F, '$1 > 7 {print $5}' iris.csv
9. awk -F, 'NR>1 {printf "%d\t%d\t%d\t%d\t%s\n", $1, $2, $3, $4, $5}' iris.csv
10. tail -n +2 iris.csv | shuf >> iris_losowe.csv
